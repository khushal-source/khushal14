
#include <stdio.h>
#include <stdlib.h>

void bubble(int arr[],int n){
    for(int i=0;i<n-1;i++){
        for(int j=0;j<n-i-1;j++){
            if(arr[j]>arr[j+1]){
                int temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
        }
    }
    // printf("\n");
    // printf("SORTED ARRAY BY BUBBLE SORT IS: ");
    for(int i=0;i<n;i++){
        printf(" %d ",arr[i]);
    }
    printf("\n");
}
void selection(int arr[],int n){
    for(int i=0;i<n-1;i++){
        int mini=999;
        int minIndex=-1;
        for(int j=i;j<n;j++){
            if(mini>arr[j]){
                mini=arr[j];
                minIndex=j;
            }
        }
        int temp=arr[i];
        arr[i]=arr[minIndex];
        arr[minIndex]=temp;
    }
    // printf("SORTED ARRAY BY SELECTION SORT IS: ");
    for(int i=0;i<n;i++){
        printf(" %d ",arr[i]);
    }
    printf("\n");
}

int main() {
    int n;
    printf("enter the size of array: ");
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        printf("enter the elements of array: ");
        scanf("%d",&arr[i]);
    }
    while(1)
    {   printf("1. FOR BUBBLE SORT\n2. FOR SELECTION SORT\n3. exit\n");

        case 1:

        case 2:

            printf("\n");

        case 3:

            exit(0);

        default:
