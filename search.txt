
#include <stdio.h>
#include <stdlib.h>

int linear(int arr[], int n, int target) {
    for (int i = 0; i < n; i++) {
        if (target == arr[i]) 
            return i;
    }
    return -1;
}

int binary(int arr[], int n, int target) {
    int temp;

    // sort the array using bubble sort
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp       = arr[j];
                arr[j]     = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    printf("\nArray elements are: ");
    for (int i = 0; i < n; i++) {
        printf(" %d ", arr[i]);
    }

    int lo = 0;
    int hi = n - 1;

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        if (arr[mid] == target)
            return mid;
        else if (arr[mid] > target)
            hi = mid - 1;
        else
            lo = mid + 1;
    }

    return -1;
}
